#!/usr/bin/env bash
# Proxmox Community Scripts - Database Installer
# Interactive installer for 12 database systems

set -e

YW="\033[33m"
BL="\033[36m"
RD="\033[01;31m"
GN="\033[1;92m"
CL="\033[m"

msg_info() {
    echo -e "${BL}[INFO]${CL} ${GN}$1${CL}"
}

msg_error() {
    echo -e "${RD}[ERROR]${CL} $1"
}

header_info() {
    cat <<"EOF"
    ____        __             __                     
   / __ \____ _/ /_____ ______/ /_  ____ _________  __
  / / / / __ `/ __/ __ `/ ___/ __ \/ __ `/ ___/ _ \/ /
 / /_/ / /_/ / /_/ /_/ / /__/ /_/ / /_/ /__  /  __/__ 
/_____/\__,_/\__/\__,_/\___/_.___/\__,_/____/\___/____/
                                                       
 Community Scripts - Database Installation
EOF
}

clear
header_info
echo ""

msg_info "Available Databases:"
echo ""
echo "  1) PostgreSQL        - Powerful relational DB"
echo "  2) MariaDB           - MySQL fork, very popular"
echo "  3) MongoDB           - NoSQL document database"
echo "  4) Redis             - In-memory key-value store"
echo "  5) InfluxDB          - Time-series database"
echo "  6) Neo4j             - Graph database"
echo "  7) Apache CouchDB    - NoSQL with HTTP API"
echo "  8) SQL Server 2022   - Microsoft SQL Server"
echo "  9) PocketBase        - SQLite backend (Go)"
echo " 10) Meilisearch       - Search engine"
echo " 11) MinIO             - S3-compatible storage"
echo " 12) Garage            - Distributed S3 storage"
echo ""

read -rp "Your choice: " choice

install_postgresql() {
    msg_info "Installing PostgreSQL..."
    bash -c "$(curl -fsSL https://raw.githubusercontent.com/community-scripts/ProxmoxVE/main/ct/postgresql.sh)"
}

case $choice in
    1) install_postgresql ;;
    2) install_mariadb ;;
    3) install_mongodb ;;
    *) msg_error "Invalid selection!" ;;
esac

echo ""
msg_info "Installation complete!"
